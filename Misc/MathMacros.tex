\newcommand{\compare}[2]{
		\ifnum #1 > #2
			+1
		\else
			\ifnum #1 < #2
				-1
			\else
				0
			\fi
		\fi
}

%-----------------------------------------
\newcommand{\signo}[1]{
	\ifnum \#1 > 0
		1
	\else
		\ifnum \#1 < 0
			-1
		\else
			0
		\fi
	\fi
}
